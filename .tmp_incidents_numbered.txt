 120: function toLabel(value: string): string {
 121:   return value
 122:     .split("_")
 123:     .map((segment) => segment.charAt(0).toUpperCase() + segment.slice(1))
 124:     .join(" ");

 127: function formatDateTime(value: string | null): string {
 128:   if (!value) {
 129:     return "N/A";
 130:   }
 131:   const parsed = new Date(value);
 132:   if (Number.isNaN(parsed.getTime())) {
 133:     return value;
 134:   }
 135:   return parsed.toLocaleString();
 136: }

 138: function defaultImpactForSeverity(severity: IncidentSeverity): IncidentServiceHealth {
 139:   if (severity === "critical") {

 190: async function readApiError(response: Response): Promise<string> {
 191:   try {
 192:     const payload = (await response.json()) as { error?: string };
 193:     if (payload.error) {
 194:       return payload.error;
 195:     }
 196:   } catch {
 197:     // ignored
 198:   }
 199:   return response.statusText || `Request failed (${response.status})`;
 200: }
