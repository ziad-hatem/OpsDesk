 138: function toLabel(value: string): string {
 139:   return value
 140:     .split("_")
 141:     .map((segment) => segment.charAt(0).toUpperCase() + segment.slice(1))
 142:     .join(" ");
 143: }

 145: function formatDateTime(value: string | null): string {
 146:   if (!value) {
 147:     return "N/A";
 148:   }
 149:   const parsed = new Date(value);
 150:   if (Number.isNaN(parsed.getTime())) {
 151:     return value;
 152:   }
 153:   return parsed.toLocaleString();
 154: }

 214: async function readApiError(response: Response): Promise<string> {
 215:   try {
 216:     const payload = (await response.json()) as { error?: string };
 217:     if (payload.error) {
 218:       return payload.error;
 219:     }
 220:   } catch {
 221:     // ignored
 222:   }
 223:   return response.statusText || `Request failed (${response.status})`;
 224: }
